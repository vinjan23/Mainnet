### service
```
sudo tee /etc/systemd/system/medasdigitald.service > /dev/null << EOF
[Unit]
Description=medasdigital
After=network-online.target
[Service]
User=$USER
ExecStart=$(which medasdigitald) start
Restart=always
RestartSec=3
LimitNOFILE=65535

[Install]
WantedBy=multi-user.target
EOF
```
```

[Unit]
```
